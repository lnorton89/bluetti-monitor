```typescript
import type { DeviceState } from '../store/ws';

/**
 * Get numeric field from device state, returning null if unavailable or invalid.
 */
function getField(state: DeviceState, field: string): number | null {
  const raw = state[field]?.value;
  if (raw === undefined) return null;
  const parsed = parseFloat(raw);
  return Number.isFinite(parsed) ? parsed : null;
}

/**
 * Format minutes as "Xh Ym" or "Ym" string.
 * Returns "—" if minutes is null, negative, or insufficient for calculation.
 */
export function formatDuration(minutes: number | null): string {
  if (minutes === null || !Number.isFinite(minutes) || minutes < 1) {
    return '—';
  }
  const hours = Math.floor(minutes / 60);
  const mins = Math.round(minutes % 60);
  if (hours > 0) {
    return mins > 0 ? `${hours}h ${mins}m` : `${hours}h`;
  }
  return `${mins}m`;
}

/**
 * Estimate runtime remaining in minutes.
 * Formula: (remaining_capacity_wh / ac_output_power_w) * 60
 * Returns null if insufficient data.
 */
export function estimateRuntimeMinutes(state: DeviceState): number | null {
  const remainingWh = getField(state, 'remaining_capacity');
  const acOutputW = getField(state, 'ac_output_power');
  const dcOutputW = getField(state, 'dc_output_power');
  
  if (remainingWh === null || remainingWh <= 0) return null;
  
  const totalOutputW = (acOutputW ?? 0) + (dcOutputW ?? 0);
  if (totalOutputW <= 0) return null;
  
  // Account for Bluetti's ~5% reserve below reported 0%
  const usableWh = Math.max(0, remainingWh - 5); // 5Wh reserve estimate
  
  return (usableWh / totalOutputW) * 60;
}

/**
 * Check if the system is currently charging from grid.
 */
export function isChargingFromGrid(state: DeviceState): boolean {
  const gridCharge = state['grid_charge_on']?.value;
  const acCharge = state['ac_charging_on']?.value;
  return gridCharge === 'True' || gridCharge === 'true' || 
         gridCharge === '1' || gridCharge === 'ON' ||
         acCharge === 'True' || acCharge === 'true' ||
         acCharge === '1' || acCharge === 'ON';
}

/**
 * Estimate time to full charge in minutes.
 * Formula: ((battery_capacity_wh - remaining_capacity_wh) / ac_input_power_w) * 60
 * Returns null if insufficient data or already full.
 */
export function estimateChargeTimeMinutes(state: DeviceState): number | null {
  const remainingWh = getField(state, 'remaining_capacity');
  const capacityWh = getField(state, 'battery_capacity');
  const acInputW = getField(state, 'ac_input_power');
  
  if (remainingWh === null || capacityWh === null) return null;
  if (remainingWh >= capacityWh) return null; // Already full
  
  const deficitWh = capacityWh - remainingWh;
  const totalInputW = acInputW ?? 0;
  
  // Also check DC input (solar charging)
  const dcInputW = getField(state, 'dc_input_power') ?? 0;
  const solarInputW = getField(state, 'solar_power') ?? 0;
  const totalChargeW = totalInputW + dcInputW + solarInputW;
  
  if (totalChargeW <= 0) return null;
  
  return (deficitWh / totalChargeW) * 60;
}

/**
 * Get battery percentage (0-100).
 */
export function getBatteryPercent(state: DeviceState): number | null {
  return getField(state, 'total_battery_percent') ?? getField(state, 'battery_percent');
}

/**
 * Check if battery is at 100%.
 */
export function isBatteryFull(state: DeviceState): boolean {
  const percent = getBatteryPercent(state);
  return percent !== null && percent >= 100;
}
```
